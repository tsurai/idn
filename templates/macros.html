{%- macro pos(data) -%} {%- if data | first | lower -%} {{ data | first}}{{data | last}} {%- elif data | last | lower -%} {%- if data | length == 2 -%} {{- data | first | first -}} {{data | first | nth(n=1)}}{{data | last}} {%- else -%} {{- self::pos(data=data | slice(end=data | length - 1)) -}} {{data | last}} {%- endif -%} {%- else -%} {%- for d in data -%} {{- self::pos(data=d) -}} {%- endfor -%} {%- endif -%} {%- endmacro pos -%}
